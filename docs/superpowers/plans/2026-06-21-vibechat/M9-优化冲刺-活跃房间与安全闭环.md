STALL_SEC = 45          # 冷场阈值
REACT_PROB = 0.4        # 真人发言后房客接话概率（DEMO 模式上调）
_watchdogs: dict[str, asyncio.Task] = {}   # slug -> 单一 watchdog

async def _maybe_react_then_arm(slug, room_id, room_slug, vibe):
    """真人发言后：按概率触发房客接话；随后（重新）挂一个冷场 watchdog。"""
    _watchdogs.pop(slug, None)  # 取消旧的，重置计时

        await _emit_tenant(slug, room_id, room_slug, vibe, kind="react")

async def _stall(slug, room_id, room_slug, vibe):

        await asyncio.sleep(STALL_SEC)
        await _emit_tenant(slug, room_id, room_slug, vibe, kind="nudge")  # 冷场救场
    except asyncio.CancelledError:

- [ ] **Step 2: 实现**（复用 `mood_field.room_mood`）

```python
from ..services.mood_field import room_mood   # 注意避免循环 import（mood_field 无反向依赖 matching，安全）

def dynamic_anchor(static: list[float], live: list[list[float]], w: float = 0.3) -> list[float]:
    if not live:
        return list(static)
    rm = room_mood(live)
    return [static[i] * (1 - w) + rm[i] * w for i in range(4)]

def complement_score(u: list[float], anchor: list[float]) -> float:
    # valence(0) 反相互补 + social(3) 同向（都想说话）+ arousal(1) 相近
    v_comp = 1.0 - abs(u[0] + anchor[0]) / 2.0     # 一负一正 → 接近 1
    s_comp = min(u[3], anchor[3])                   # 都愿说话
    a_comp = 1.0 - abs(u[1] - anchor[1])
    return v_comp * 0.5 + s_comp * 0.3 + a_comp * 0.2
```

- [ ] **Step 3: `match_rooms` 升级签名**——接收 `live_by_slug: dict[str, list[list[float]]]`，每房用 `dynamic_anchor`；额外返回 top 互补房。

```python
def match_rooms(user_vec, rooms, live_by_slug=None):
    live_by_slug = live_by_slug or {}
    scored = [{**rm, "affinity": affinity(user_vec,
                dynamic_anchor(rm["anchor_vector"], live_by_slug.get(rm["slug"], []))),
              "_sim": cosine_sim(user_vec, dynamic_anchor(...))} for rm in rooms]
    scored.sort(key=lambda x: x["_sim"], reverse=True)
    # 互补房：在 valence 与用户相反的房里取 complement_score 最高
    cands = [rm for rm in scored if rm.get("anchor_vector") and (rm["anchor_vector"][0] * user_vec[0] < 0)]
    comp = max(cands, key=lambda rm: complement_score(user_vec, rm["anchor_vector"]), default=None)
    for x in scored: x.pop("_sim", None)
    return scored, comp
```
